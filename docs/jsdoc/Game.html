class Game extends Impacto.Scene {
	start() {
		this.rect = new Impacto.GameObjects.Rectangle(400, 300, 100, 100, "#ff0000");
		this.addChild(this.rect);
	}
}

const game = new Impacto.Game({
	width: 800,
	height: 600,
	backgroundColor: "#f0f0f0",
	scene: Game,
});
